jqTree

JqTree is a tree widget written in jQuery UI.

  • Create a tree from JSON data.
  • Drag and drop.
var data = [
	{
		label: 'node1',
		children: [
			{ label: 'child1' },
			{ label: 'child2' }
		]
	},
	{
		label: 'node2',
		children: [
			{label: 'child3'}
		]
	}
];
$('#tree1').tree({
	data: data,
	autoOpen: true,
	dragAndDrop: true
});
